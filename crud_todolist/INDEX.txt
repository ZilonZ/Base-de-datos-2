  Response: { "success": true, "data": [...] }

UPDATE (Actualizar)
  POST /api_update.php
  Body: { "id": 1, "title": "Nuevo", "completed": true }
  Response: { "success": true, "data": {...} }

DELETE (Eliminar)
  POST /api_delete.php
  Body: { "id": 1 }
  Response: { "success": true }

❌ SOLUCIÓN DE PROBLEMAS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

"Cannot POST /api_create.php"
  → Apache no está corriendo
  → Carpeta no está en C:\xampp\htdocs\
  → Recarga la página (Ctrl+F5)

"Connection refused" (BD)
  → PostgreSQL no está corriendo
  → Inicia PostgreSQL en XAMPP
  → Espera 5 segundos

"FATAL: database does not exist"
  → Crea la BD "todolist" en pgAdmin
  → Ejecuta database.sql

"No se ven cambios"
  → Abre consola (F12) y mira los errores
  → Revisa que PostgreSQL esté corriendo
  → Recarga la página
